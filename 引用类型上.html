Object 类型

创建Object实例的方式有两种。第一种是使用new操作符后跟object构造函数。

		
			var person = new Object();
			person.name = "zsw";
			person.age = 27;
			document.write(person.name + "\n" + person.age + "\n"); //zsw 27

另一种方式是字符字面量表示法

		
			var per = {
				name: "zsw";
				age: 27;
			}
			document.write(per.name + "\n" + per.age + "\n");

对象字面量向函数传递大量可选参数

		
			function displayInfo(args) {
				var output = "";

				if (typeof args.name == "string") {
					output += "Name: " + args.name + "\n";
				}
				if (typeof args.age == "Number") {
					output += "Age: " + args.age + "\n";
				}
				document.write(output + "\n");
			}

			displayInfo({
				name: "zsw",
				age: 27
			});
			displayInfo({
				name: "Greg"
			})

属性名可以是字符串或者数字，但是如果以数字作为属性名，则必须以方括号（方括号记法）来获得这个数字属性名对应的属性值。

		
			var ageGroup = {
				30: "Children",
				100: "Very Old"
			}
			document.write(ageGroup.30); //undefined
			document.write(ageGroup["30"]); //Children

Array类型

创建数组的基本方式有两种。第一种是使用Array构造函数

		
			var colors = new Array();

第二种是数组字面量表示法

		
			var colors = ["red", "blue", "green"];

数组转换方法

调用数组的toString()方法会返回数组中每个值得字符串形式拼接而成的一个以逗号分隔的字符串。

		
			var colors = ["red","blue","green"];
			document.write(colors.toString()); //red, blue,green

调用数组的toLocaleString()方法

		
			var person1 = {
				toLocaleString : function() {
					return "zsw";
				}

				toString : function() {
					return "nik";
				}
			};
			var person2 = {
				toLocaleString : function() {
					return "gir"
				}	 
				toString : function() {
					return "gre"
				} 
			};
			var people = [person1, person2];
			document.write(people); //"nik","gre"
			document.write(people.toString()); //"nik","gre"
			document.write(people.toLocaleString()); //"zsw","gir"

数组的使用方法

栈方法：push()和pop()的方法(访问规则是先进先出) push()方法可以接收任何参数，逐个添加到数组末尾，并返回修改后数组的长度。pop()方法则从数组末尾移除最后一项，减少数组的length值，然年返回移除的项。

		
			var a = ["a","b","c"];
			var count = a.push("d","e");
			document.write(a); //5;
			document.write(a);

			var item = a.pop();
			document.write(item); //"e"
			document.write(a); //"a","b","c","d"

队列方法：shift()和unshift()方法(访问规则是后进先出) shift()方法移除数组第一项并返回该项，同时将数组长度减一。unshift()方法在数组前端添加任意个项并返回新的数组长度。

		
			var abbr = ["red","blue","green"];
			var count = abbr.shift();
			document.write(count); //red
			document.write(abbr); //"blue","green"

			var item = abbr.unshift("red");
			document.write(item); //3
			document.write(abbr); //"red","blue","green"

重排序方法

reverse()和sort()方法 reverse()方法会反转数组的顺序。sort()方法按升序排列数组项。

		
			var value = [1,2,3,4,5];
			value.reverse();
			document.write(value); //5,4,3,2,1
		
	
		
			var value_b = [5,4,3,2,1];
			value_b.sort();
			document.write(value_b); //1,2,3,4,5

sort()最好的方式

		
			function compare(value1,value2) {
				if (value1 < value2) {
					return -1;
				}else if (value1 > value2) {
					return 1;
				}else {
					return 0;
				}
			}
			var value_c = [20,5,10,1,15];
			document.write(value_c.sort(compare)); //1,5,10,15,20

reverse()最好的方式

		
			function compare_a(num1,num2) {
				if (num1 < num2) {
					return 1;
				}else if (num1 > num2) {
					return -1;
				}else {
					return 0;
				}
			}
			var value_d = [20,5,10,1,15];
			document.write(value_d.reverse(compare_a)); //20,15,10,5,1

操作方法

concat()可以基于当前数组中的所有项创建一个新数组。

		
			var string = ["apple", "orange", "banana"];
			var string2 = string.concat("peach","lemon");
			document.write(string2); //"apple", "orange", "banana","peach","lemon"
			document.write(string); //"apple", "orange", "banana"

slice()方法，它能够基于当前数组中的一个项或多个项创建一个新数组。可已接收两个参数，起始项和结束位置 结束位置不包括结束项

		
			var string3 = ["apple","orange","banana","peach","lemon"];
			var string4 = string3.slice(1);
			document.write(string4);//"orange","banana","peach","lemon"
			var string5 = string3.slice(2,4); 
			document.write(string4); //"banana","peach"

splice()方法 包含3个参数：起始位置，要删除的项数，要插入的项数。

		
			var color_a = ["red","green","blue"];
			var removed = color_a.splice(0,1);
			document.write(color_a); //"green","blue";
			document.write(removed); //"red";

			removed = color_a.splice(1,0,"yellow","orange");
			document.write(color_a); // "green","blue","yellow","orange"
			document.write(removed); // 返回空数组

			removed = color_a.splice(1,1,"red","purple");
			document.write(color_a); // "green","red","purple","yellow","orange"
			document.write(removed); // blue
